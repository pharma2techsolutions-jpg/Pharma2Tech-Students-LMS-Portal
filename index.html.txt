<script>
    const apiURL = "https://script.google.com/macros/s/AKfycby9tnwCHXFUG7QXLtvz-PknytYrSDG_VosF0InTweuKvRmp5nrzeXFZX6N0QE9rPLND/exec";
    let userEmail = "";

    async function handleLogin() {
        const e = document.getElementById('email').value.trim();
        const p = document.getElementById('pass').value.trim();
        if(!e || !p) return alert("Please fill all fields.");

        toggleLoader(true);
        try {
            const res = await fetch(`${apiURL}?action=login&email=${encodeURIComponent(e)}&pass=${encodeURIComponent(p)}`);
            const d = await res.json();
            if(d.status === "SUCCESS") {
                userEmail = d.email;
                document.getElementById('studentNameDisplay').innerText = "Hello, " + d.name;
                document.getElementById('loginPage').style.display = "none";
                document.getElementById('dashboard').style.display = "flex";
                document.body.style.background = "#f4f7f6";
                loadData();
            } else { alert("Invalid Credentials!"); }
        } catch(err) { alert("Server error."); }
        toggleLoader(false);
    }

    async function loadData() {
        const res = await fetch(`${apiURL}?action=getDashboardData&email=${userEmail}`);
        const data = await res.json();

        // Attendance Logic: Absent asel tarach Watch Video disel
        document.getElementById('attBody').innerHTML = data.attendance.slice(1).map(r => {
            const status = r[2]; 
            const actionBtn = status === 'Absent' 
                ? `<a href="${r[4]}" target="_blank" class="action-link">Watch Video</a>` 
                : `<span style="color:#28a745; font-weight:600;"><i class="fas fa-check-circle"></i> Attended</span>`;
            return `<tr><td>${r[1]}</td><td><b style="color: ${status === 'Absent' ? 'var(--danger)' : 'var(--success)'}">${status}</b></td><td>${r[3]}</td><td>${actionBtn}</td></tr>`;
        }).join('');

        // Exams
        document.getElementById('examBody').innerHTML = data.exams.slice(1).map(r => {
            const status = (r[6] || "").toUpperCase();
            const score = status === "UPCOMING" ? "--" : `${r[3]}/${r[4]}`;
            const action = status === "UPCOMING" ? `<a href="${r[7]}" target="_blank" class="action-link" style="background:var(--success)">Start Exam</a>` : `<span style="color:#999">Completed</span>`;
            return `<tr><td>${r[2]}</td><td>${score}</td><td>${r[5]}</td><td><span class="status-pill ${status}">${status}</span></td><td>${action}</td></tr>`;
        }).join('');

        // Assignments
        document.getElementById('assignBody').innerHTML = data.assignments.slice(1).map(r => `<tr><td>${r[1]}</td><td>${r[2]}</td><td><b>${r[3]}</b></td></tr>`).join('');
    }

    function showTab(id, el) {
        document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
        document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
        document.getElementById(id).classList.add('active');
        el.classList.add('active');
    }

    function logout() { if(confirm("Logout?")) location.reload(); }
    function toggleLoader(show) { document.getElementById('loader').style.display = show ? "flex" : "none"; }
</script>
